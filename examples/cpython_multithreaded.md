```python
import thread, time
import numpy

class A:
	def __init__(self, name):
		self.name  = name
		self.value = 0
		self.flag  = True
		self.number = 1.5
		self.array = numpy.array([1,2,3,4]).astype(int)

	def echo(self, s):
		print s

	def show_array(self):
		print self.array

	def run(self):
		print '-----starting run'
		while self.value < 10:
			print 'from cpython:', self.value
			self.value += 1
			time.sleep(0.1)
		print '------end run'
	def run_thread(self):
		thread.start_new_thread(self.run, ())


```
